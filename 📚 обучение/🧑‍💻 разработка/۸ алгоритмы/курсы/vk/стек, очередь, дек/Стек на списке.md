```go
package main  
  
type Stack struct {  
    head *Node  
}  
  
type Node struct {  
    data int  
    next *Node  
}  
  
func (s *Stack) Push(data int) {  
    node := &Node{  
       data: data,  
       next: s.head,  
    }  
    s.head = node  
}  
  
func (s *Stack) Pop() int {  
    // need to move head to next  
    if s.head == nil {  
       return 0 // default value  
    }  
  
    val := s.head.data  
    s.head = s.head.next  
  
    return val  
}  
  
func (s *Stack) Peek() int {  
    if s.head == nil {  
       return 0  
    }  
  
    return s.head.data  
}
```
