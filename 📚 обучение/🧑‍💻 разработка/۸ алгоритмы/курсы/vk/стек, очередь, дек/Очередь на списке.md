```go
package queue  
  
// need to write queue on stack  
var _ Queue = &queue{}  
  
type Queue interface {  
    Append(data int)  
    Remove() int  
}  
  
// main idea head is
type queue struct {  
    head *Node  
    tail *Node  
}  
  
func NewQueue() *queue {  
    return new(queue)  
}  
  
type Node struct {  
    data int  
    next *Node  
}  
  
func (q *queue) Append(data int) {  
    newNode := &Node{data: data}  
    if q.head == nil {  
       q.head = newNode  
       q.tail = newNode  
       return  
    }  
  
    q.tail.next = newNode  
    q.tail = newNode  
}  
  
func (q *queue) Remove() int {  
    if q.head == nil {  
       return 0  
    }  
  
    val := q.head.data  
    q.head = q.head.next  
  
    if q.head == nil {  
       q.tail = nil  
    }  
  
    return val  
}
```
