```txt
В исходную строку добавили некоторое количество символов. Необходимо выявить, была ли строка **а** исходной для строки **b**.
```

// через очередь
```go
func isSubsequence(a, b string) bool {  
    q := queue.NewQueue[rune]()  
    for _, el := range a {  
       q.Push(el)  
    }  
  
    for _, el := range b {  
       if q.First() == el {  
          q.Pop()  
       }  
    }  
  
    return q.IsEmpty()  
}
```

// через указатель
```go
func isSubsequenceByTwoPointers(a, b string) bool {  
    if len(a) == 0 {  
       return true  
    }  
  
    aRune := []rune(a)  
  
    i := 0  
    for _, el := range b {  
       if el == aRune[i] {  
          i++  
  
          if len(a) == i {  
             return true  
          }  
       }  
    }  
  
    return false  
}
```
