```go
type LinkedList struct {  
    head *Node  
}  
  
type Node struct {  
    next *Node  
    data int  
}  
  
func RmElemFromList(list *LinkedList, val int) {  
    dummy := &Node{next: list.head}  
  
    prev := dummy  
    curr := list.head  
  
    for curr != nil {  
       if curr.data != val {  
          prev = curr  
       } else {  
          prev.next = curr.next  
       }  
  
       curr = curr.next  
    }  
  
    list.head = dummy.next  
}
```
