package dequeue  
  
type Deque interface {  
    Push(data int)  
    Pop() int  
    Append(data int)  
    Remove() int  
}  
  
type deque struct {  
    head *Node  
    tail *Node  
}  
  
type Node struct {  
    data int  
    prev *Node  
    next *Node  
}  
  
func (dq *deque) Push(data int) {  
    newNode := &Node{  
       data: data,  
       next: dq.head,  
    }  
  
    dq.head = newNode  
}  
  
func (dq *deque) Pop() int {  
    if dq.head == nil {  
       return 0  
    }  
  
    val := dq.head.data  
    dq.head = dq.head.next  
  
    if dq.head != nil {  
       dq.head.prev = nil  
    } else {  
       dq.tail = nil  
    }  
  
    return val  
}  
  
func (dq *deque) Append(data int) {  
    newNode := &Node{  
       data: data,  
       prev: dq.tail,  
    }  
  
    if dq.tail == nil {  
       dq.head = newNode  
       dq.tail = newNode  
       return  
    }  
  
    dq.tail.next = newNode  
    dq.tail = newNode  
}  
  
// Remove from end.func (dq *deque) Remove() int {  
    if dq.tail == nil {  
       return 0  
    }  
  
    val := dq.tail.data  
    dq.tail = dq.tail.prev  
  
    if dq.tail == nil {  
       dq.head = nil  
    } else {  
       dq.tail.next = nil  
    }  
  
    return val  
}
